Alternative methods for chemical equilibrium¶

The methods previously examined for determining the equilibrium composition rely on knowing the chemical reaction(s) occurring, and can involve highly nonlinear equations.

Fortunately, we have methods that do not require knowing what reaction(s) are occurring. We will compare two such solution methods:

  1. Direct minimization of Gibbs free energy

  2. Lagrange’s method of undetermined multipliers

import numpy as np
import cantera as ct
from scipy.optimize import root, Bounds, minimize, basinhopping

from pint import UnitRegistry
ureg = UnitRegistry()
Q_ = ureg.Quantity

# for convenience:
def to_si(quant):
    '''Converts a Pint Quantity to magnitude at base SI units.
    '''
    return quant.to_base_units().magnitude

Solution via direct minimization of Gibbs free energy¶

def calc_total_gibbs(x, temperature, pressure, components, gas, moles_init):
    '''System of equations for reaction coordinate and equilibrium composition.
    '''
    total_moles = x[0]
    mole_fractions = np.array(x[1:])
    
    moles = mole_fractions * total_moles
    
    # get standard-state Gibbs free energy of each component
    gibbs = np.zeros(len(components))
    for idx, comp in enumerate(components):
        gas.TPX = (
            to_si(temperature), to_si(Q_(1, 'atm')),
            f'{comp}:1.0'
            )
        gibbs[idx] = gas.gibbs_mole
        
    gibbs *= Q_('J/kmol')
    
    gas_constant = Q_(ct.gas_constant, 'J/(kmol*K)')
    chemical_potentials = (
        gibbs + gas_constant * temperature * np.log(
            mole_fractions * pressure / Q_(1.0, 'atm')
            )
        )
    
    return to_si(np.sum(moles * chemical_potentials))

def equality_cons(x, temperature, pressure, components, gas, moles_init):
    total_moles_init = np.sum(moles_init)
    return np.array([
        x[0] * (x[1] + 2*x[2] + 2*x[3]) - 
        (moles_init[0] + 2*moles_init[1] + 2*moles_init[2]),
        x[0] * (x[1] + x[3]) - (moles_init[0] + moles_init[2])
        ])

def equality_cons_jac(x, temperature, pressure, components, gas, moles_init):
    total_moles_init = np.sum(moles_init)
    return np.array([
        [x[1], x[0], 2*x[0], 2*x[0]],
        [x[1], x[0], 0.0, x[0]]
        ])

Solution via Lagrange method of undetermined multipliers¶

def lagrange_system(x, temperature, pressure, components, gas, moles_init):
    '''System of equations for reaction coordinate and equilibrium composition.
    '''
    moles = np.array([x[0], x[1], x[2]])
    multipliers = np.array([x[3], x[4]]) * Q_('J/kmol')
    
    mole_fractions = moles / np.sum(moles)
    
    # get standard-state Gibbs free energy of each component
    gibbs = np.zeros(len(components))
    for idx, comp in enumerate(components):
        gas.TPX = (
            to_si(temperature), to_si(Q_(1, 'atm')),
            f'{comp}:1.0'
            )
        gibbs[idx] = gas.gibbs_mole
        
    gibbs *= Q_('J/kmol')
    
    gas_constant = Q_(ct.gas_constant, 'J/(kmol*K)')
    chemical_potentials = (
        gibbs + gas_constant * temperature * np.log(
            mole_fractions * pressure / Q_(1.0, 'atm')
            )
        )
    
    moles_initial_C = moles_init[0] + moles_init[2]
    moles_initial_O = moles_init[0] + 2*moles_init[1] + 2*moles_init[2]
    
    moles_C = moles[0] + moles[2]
    moles_O = moles[0] + 2*moles[1] + 2*moles[2]
    
    return [
        moles_C - moles_initial_C,
        moles_O - moles_initial_O,
        to_si(chemical_potentials[0] + np.sum(multipliers * [1, 1])),
        to_si(chemical_potentials[1] + np.sum(multipliers * [0, 2])),
        to_si(chemical_potentials[2] + np.sum(multipliers * [1, 2])),
        ]

components = ['CO', 'O2', 'CO2']
moles_initial = np.array([1.0, 0.5, 0.0])

temperature = Q_(2500, 'K')
